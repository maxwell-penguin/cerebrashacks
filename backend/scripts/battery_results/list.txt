import { useState } from 'react';

function App() {
  const [search, setSearch] = useState('');

  const users = [
    { name: 'Alice Smith', email: 'alice@example.com', role: 'Admin' },
    { name: 'Bob Jones', email: 'bob@example.com', role: 'Editor' },
    { name: 'Carol Lee', email: 'carol@example.com', role: 'Viewer' },
    { name: 'Dan Wu', email: 'dan@example.com', role: 'Editor' },
  ];

  const filtered = users.filter(
    (u) =>
      u.name.toLowerCase().includes(search.toLowerCase()) ||
      u.email.toLowerCase().includes(search.toLowerCase()) ||
      u.role.toLowerCase().includes(search.toLowerCase())
  );

  return (
    <section className="container mx-auto p-4 flex flex-col space-y-4">
      <h1 className="text-2xl font-bold mb-2">Users</h1>

      <input
        type="text"
        placeholder="Search..."
        value={search}
        onChange={(e) => setSearch(e.target.value)}
        className="w-full md:w-1/2 p-2 border rounded focus:outline-none focus:ring-2 focus:ring-blue-500"
      />

      <div className="overflow-x-auto rounded shadow-sm">
        <table className="w-full table-auto border-collapse bg-white">
          <thead className="bg-gray-100">
            <tr>
              <th className="p-2 text-left">Name</th>
              <th className="p-2 text-left">Email</th>
              <th className="p-2 text-left">Role</th>
            </tr>
          </thead>
          <tbody>
            {filtered.map((user, idx) => (
              <tr
                key={idx}
                className={`border-t ${idx !== filtered.length - 1 ? '' : ''} hover:bg-gray-50`}
              >
                <td className="p-2">{user.name}</td>
                <td className="p-2">{user.email}</td>
                <td className="p-2">{user.role}</td>
              </tr>
            ))}
            {filtered.length === 0 && (
              <tr>
                <td colSpan="3" className="p-2 text-center text-gray-500">
                  No users found.
                </td>
              </tr>
            )}
          </tbody>
        </table>
      </div>
    </section>
  );
}

export default App;
