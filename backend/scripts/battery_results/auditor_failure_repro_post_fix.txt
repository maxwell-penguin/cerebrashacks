import { useState } from 'react';

export default function App() {
  const users = [
    { name: 'Alice Smith', email: 'alice@example.com', role: 'Admin' },
    { name: 'Bob Jones', email: 'bob@example.com', role: 'Editor' },
    { name: 'Carol Lee', email: 'carol@example.com', role: 'Viewer' },
    { name: 'Dan Wu', email: 'dan@example.com', role: 'Editor' },
  ];

  const [search, setSearch] = useState('');

  const filteredUsers = users.filter(({ name, email, role }) => {
    const term = search.toLowerCase();
    return (
      name.toLowerCase().includes(term) ||
      email.toLowerCase().includes(term) ||
      role.toLowerCase().includes(term)
    );
  });

  return (
    <main className="p-4 md:p-6 flex flex-col gap-4">
      <header className="flex flex-col md:flex-row justify-between items-center mb-4">
        <h1 className="text-2xl font-bold mb-2 md:mb-0">Users</h1>
        <input
          type="text"
          placeholder="Search..."
          aria-label="Search users"
          value={search}
          onChange={(e) => setSearch(e.target.value)}
          className="border rounded px-2 py-1 w-full md:w-64 focus:outline-none focus:ring-2 focus:ring-blue-500"
        />
      </header>

      <section className="overflow-x-auto rounded shadow-sm">
        <table className="min-w-full border border-gray-200">
          <thead className="bg-gray-100">
            <tr>
              <th className="px-4 py-2 text-left font-medium">Name</th>
              <th className="px-4 py-2 text-left font-medium">Email</th>
              <th className="px-4 py-2 text-left font-medium">Role</th>
            </tr>
          </thead>
          <tbody>
            {filteredUsers.map((user, idx) => (
              <tr
                key={idx}
                className="border-t hover:bg-gray-50 transition-colors"
              >
                <td className="px-4 py-2">{user.name}</td>
                <td className="px-4 py-2">{user.email}</td>
                <td className="px-4 py-2">{user.role}</td>
              </tr>
            ))}
            {filteredUsers.length === 0 && (
              <tr>
                <td
                  colSpan="3"
                  className="px-4 py-2 text-center text-gray-500"
                >
                  No users found.
                </td>
              </tr>
            )}
          </tbody>
        </table>
      </section>
    </main>
  );
}
